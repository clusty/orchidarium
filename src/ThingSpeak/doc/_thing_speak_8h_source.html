1040  {

1042  bool fFirstItem = true;
1043  for(size_t iField = 0; iField < 8; iField++)
1044  {
1045  if(this->nextWriteField[iField].length() > 0)
1046  {
1047  if(!fFirstItem)
1048  {
1049  postMessage = postMessage + String("&");
1050  }
1051  postMessage = postMessage + String("field") + String(iField + 1) + String("=") + this->nextWriteField[iField];
1052  fFirstItem = false;
1053  this->nextWriteField[iField] = "";
1054  }
1055  }
1056 
1057  if(!isnan(nextWriteLatitude))
1058  {
1059  if(!fFirstItem)
1060  {
1061  postMessage = postMessage + String("&");
1062  }
1063  postMessage = postMessage + String("lat=") + String(this->nextWriteLatitude);
1064  fFirstItem = false;
1065  this->nextWriteLatitude = NAN;
1066  }
1067 
1068  if(!isnan(this->nextWriteLongitude))
1069  {
1070  if(!fFirstItem)
1071  {
1072  postMessage = postMessage + String("&");
1073  }
1074  postMessage = postMessage + String("long=") + String(this->nextWriteLongitude);
1075  fFirstItem = false;
1076  this->nextWriteLongitude = NAN;
1077  }
1078 
1079 
1080  if(!isnan(this->nextWriteElevation))
1081  {
1082  if(!fFirstItem)
1083  {
1084  postMessage = postMessage + String("&");
1085  }
1086  postMessage = postMessage + String("elevation=") + String(this->nextWriteElevation);
1087  fFirstItem = false;
1088  this->nextWriteElevation = NAN;
1089  }
1090 
1091  if(this->nextWriteStatus.length() > 0)
1092  {
1093  if(!fFirstItem)
1094  {
1095  postMessage = postMessage + String("&");
1096  }
1097  postMessage = postMessage + String("status=") + String(this->nextWriteStatus);
1098  fFirstItem = false;
1099  this->nextWriteStatus = "";
1100  }

1102  if(this->nextWriteTwitter.length() > 0)

1107  }

1111  }

1117  postMessage = postMessage + String("&");

1121  this->nextWriteTweet = "";
1122  }
1123 
1124  if(this->nextWriteCreatedAt.length() > 0)
1125  {
1126  if(!fFirstItem)
1127  {
1128  postMessage = postMessage + String("&");
1129  }
1130  postMessage = postMessage + String("created_at=") + String(this->nextWriteCreatedAt);

1187  int writeRaw(unsigned long channelNumber, String postMessage, const char * writeAPIKey)
1188  {
1189  #ifdef PRINT_DEBUG_MESSAGES
1190  Particle.publish(SPARK_PUBLISH_TOPIC, "ts::writeRaw (channelNumber: " + String(channelNumber) + " writeAPIKey: " + String(writeAPIKey) + " postMessage: \"" + postMessage + "\")" , SPARK_PUBLISH_TTL, PRIVATE);
1191  #endif
1192 
1193  if(!connectThingSpeak())
1194  {
1195  // Failed to connect to ThingSpeak
1196  return ERR_CONNECT_FAILED;
1197  }
1198 
1199  postMessage = postMessage + String("&headers=false");
1200 
1201  #ifdef PRINT_DEBUG_MESSAGES
1202  Particle.publish(SPARK_PUBLISH_TOPIC, "Post " + postMessage, SPARK_PUBLISH_TTL, PRIVATE);
1203  #endif
1204 
1205  postMessage = postMessage + String("\n");
1206 
1207  // Post data to thingspeak
1208  if(!this->client->print("POST /update HTTP/1.1\r\n")) return abortWriteRaw();
1209  if(!writeHTTPHeader(writeAPIKey)) return abortWriteRaw();
1210  if(!this->client->print("Content-Type: application/x-www-form-urlencoded\r\n")) return abortWriteRaw();
1211  if(!this->client->print("Content-Length: ")) return abortWriteRaw();
1212  if(!this->client->print(postMessage.length())) return abortWriteRaw();
1213  if(!this->client->print("\r\n\r\n")) return abortWriteRaw();
1214  if(!this->client->print(postMessage)) return abortWriteRaw();
1215 
1216  String entryIDText = String();
1217  int status = getHTTPResponse(entryIDText);
1218  if(status != OK_SUCCESS)
1219  {
1220  client->stop();
1221  return status;
1222  }
1223  long entryID = entryIDText.toInt();
1224 
1225  #ifdef PRINT_DEBUG_MESSAGES
1226  Particle.publish(SPARK_PUBLISH_TOPIC, " Entry ID \"" + entryIDText + "\" (" + String(entryID) + ")" , SPARK_PUBLISH_TTL, PRIVATE);
1227  #endif
1228 
1229  client->stop();
1230 
1231  #ifdef PRINT_DEBUG_MESSAGES
1232  Particle.publish(SPARK_PUBLISH_TOPIC, "disconnected.", SPARK_PUBLISH_TTL, PRIVATE);
1233  #endif
1234  if(entryID == 0)
1235  {
1236  // ThingSpeak did not accept the write
1237  status = ERR_NOT_INSERTED;
1238  }
1239  return status;
1240  };
